drop database if exists lcs;

use lcs;

create table tb_cards  (
id int auto_increment,
question text,
answer1 text,
answer2 text,
answer3 text,
answer4 text,
releaseDate int,
PRIMARY KEY (id));

create table tb_pulls (
id int auto_increment,
polldatetime bigint,
PRIMARY KEY (id));

create table tb_cards_done (
id int auto_increment,
card_id int,
done_datetime bigint,
correct tinyint,
PRIMARY KEY (id));
